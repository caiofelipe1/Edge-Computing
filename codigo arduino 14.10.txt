#include<Servo.h>
#define led 13
#define trigger 7
#define echo 8
#define ServoPin 9
 
int dist = 0;
 
Servo myServo;
 
void setup() {
  myServo.attach(ServoPin);
  pinMode(led,OUTPUT);
  pinMode(trigger,OUTPUT);
  pinMode(echo,INPUT);
  pinMode(ServoPin,OUTPUT);
  Serial.begin(9600);
}
 
void loop() {
  digitalWrite(trigger, LOW);
  delayMicroseconds(2);
  digitalWrite(trigger,HIGH);
  delayMicroseconds(10);
  digitalWrite(trigger,LOW);
 
  dist = pulseIn(echo,HIGH);
  dist = dist/58;
 
  if(Serial.available() > 0){
    char comando = Serial.read();
    if(comando == 'M'){
      myServo.write(90);
    }else if(comando == 'L'){
      myServo.write(180);
    }else if(comando == 'R'){
      myServo.write(0);
    }else if(comando == 'A'){
      digitalWrite(led, HIGH);
    }else if(comando = 'D'){
      digitalWrite(led,LOW);
    }
  }
  Serial.println(dist);
  delay(2000);
 
}
